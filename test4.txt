// test if / else if / else

Oświadczam powołanie do życia bytu o imieniu x który od pierwszego tchnienia niesie wartość 10;
Oświadczam powołanie do życia bytu o imieniu y który od pierwszego tchnienia niesie wartość 20;

// powinno wejść do pierwszego if
Wybaczcie śmiałość lecz rozważam czy przypadkiem (x jest 10) {
    oznajmij("x jest równe 10");
} Lecz jeśli los zechce (y jest 10) {
    oznajmij("y jest równe 10");
} Jeżeli nie to zaiste uczynię co innego {
    oznajmij("żaden warunek nie był prawdziwy");
}
